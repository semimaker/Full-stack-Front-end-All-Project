CREATE DATABASE backend_jsp_project CHARACTER SET utf8mb4;
USE backend_jsp_project;

#Create -> 데이터 베이스_테이블을 생성할 떄 사용한다.
CREATE TABLE menu(
	idx int NOT NULL AUTO_INCREMENT,
    content varchar(255) NOT NULL,
    PRIMARY KEY(idx)
);

#INSERT, UPDATE, SELECT, DELETE
# -> 테이블 내 데이터(문자,숫자,boolean,null)등을
# 수정, 조작하게 해주는 mysql 명령어.

INSERT INTO menu(content) VALUES("부모님 감사합니다.!");
#데이터삽입.
SELECT * FROM menu;
#데이터 조합
UPDATE menu SET content = "블랙굿굿." WHERE idx = 1;
