package study;

public class Java_method {
	//0.리턴 타입과 리턴 값이 없는 경우의 메소드 기능 정의(조립)
	void add2(int a, int b) {
		int result = a + b;
		System.out.println(result);
		//출력은 되나, result변수는 이 중괄호 블록 안에서만 사용.
	}
	//계산기에 기능을 정의.
	int add(int a, int b) {
		return a+b;
	}
	
	int subtract(int a, int b) {
		return a-b;
	}
	
	int multiply(int a, int b) {
		return a*b;
	}
	
	int division(int a, int b) {
		return a/b;
	}
	public static void main(String[] args) {
	
		//2. 계산기의 기능들을 실행(호출).
		Java_method calculator = new Java_method(); 
		//다른 메서드를 이용하려고 하면 그에 맞는 객체를 생성해주어야 한다.
		//계산기 클래스에서 정의한 메서드를 이용하려고 할 시 객체를 생성해줘야 한다.
		int sum = calculator.add(7,8);
		int difference = calculator.subtract(10,3);
		int product = calculator.multiply(3, 4);
		int divideResult = calculator.division(9,3);
		
		System.out.println(sum);
		System.out.println(difference);
		System.out.println(product);
		System.out.println(divideResult);
		//계산기가 실행될 떄, 우리가 정의한 기능들을 실행시킬 떄-> 기능을 호출할 떄.
		//사용할 메소드를 호출 매세드로서 작성.
	}
}

//자바 메서드 기능 정리 -> 계산기는 조립.
//자바 메서드 기능 실행 -> 가산기를 실행.
//calculator : 계산기 본판으로 메서드를 만드는 법.
