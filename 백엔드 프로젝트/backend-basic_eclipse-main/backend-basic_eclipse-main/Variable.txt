package study;

public class Variable {

	public static void main(String[] args) {
		//프로그램의 시작점.
		int a = 1;
		float b= 1.1F;
		double c = 2.22D;
		char d = '오';
		boolean e = true;
		boolean f = false;

		System.out.println(a); //콘솔에 글을 출력.
		System.out.println(b);
		System.out.println(c);
		System.out.println(d);
		System.out.println(e);
		System.out.println(f);
	}
}
