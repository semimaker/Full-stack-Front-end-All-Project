package study;

public class Person {
		
		private String name = "안녕";//멤버변수(필드)
		
		public String getName() {
			return name;
		}//다른 이름의 클래스에서 조회할 수 있게 하는 메소드.
		
		public void setName(String name) {
			this.name =name;
		}
}
