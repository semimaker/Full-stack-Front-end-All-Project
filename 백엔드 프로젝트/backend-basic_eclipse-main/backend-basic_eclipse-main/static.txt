package study;

public class statics {

	int shareWater = 12;
	//가장 바깥쪽의 변수 선언한, 전역변수.
	//어느 곳에서도 사용이 가능하다.
	
	void drinkWater(int amount) {
		shareWater = shareWater - amount;
	}//지역변수.
	
	//블록 안의 영역.
	void checkWater() {
		if(shareWater>0) {
			int y = shareWater;//블록변수.
			//전역변수인 관계로 어디서든 접근이 가능.
			//y같은 블록 변수는 해당 영역 안에서만 사용 가능.
		}
	}
	//전역변수 -> 지역변수 -> 블록변수 사용이 가능하며, 전역은 지역,블록 다 사용이 가능.
}
